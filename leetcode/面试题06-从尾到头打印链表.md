输入一个链表的头节点，从尾到头反过来返回每个节点的值（用数组返回）。

 

```
示例 1：

输入：head = [1,3,2]
输出：[2,3,1]
```

 

#### 限制：

- 0 <= 链表长度 <= 10000


#### Solutions

1. ##### recursion(system stack)

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    vector<int> res;
    void traverse(ListNode * head) {
        if (!head) return;
        traverse(head->next);
        res.push_back(head->val);
    }

    vector<int> reversePrint(ListNode* head) {
        traverse(head);
        return res;
    }
};
```


2. ##### reverse


```c++
class Solution {
public:
    vector<int> reversePrint(ListNode* head) {
        vector<int> res;
        while (head) {
            res.push_back(head->val);
            head = head->next;
        }

        reverse(res.begin(), res.end());

        return res;
    }
};
```

3. ##### reverse assignment

```c++
class Solution {
public:
    vector<int> reversePrint(ListNode* head) {
        int len = 0;
        ListNode * tmp = head;
        while (tmp) {
            len++;
            tmp = tmp->next;
        }
        vector<int> res(len);

        while (head) {
            res[--len] = head->val;
            head = head->next;
        }

        return res;
    }
};
```
