1. ##### sliding window

```cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
    int n, maxd, pos;
    cin >> n; cin >> maxd;
    vector<int> res(n);
    for (int i = 0; i < n; i++) {
        cin >> pos;
        res[i] = pos;
    }
    long cnt = 0;
    for (int i = 0, j = 0; i < n; i++) {
        while (j + 1 < res.size() && res[j + 1] - res[i] <= maxd)
            j++;
        long num = max(j - i, 1);
        cnt = (cnt + (num * (num - 1) / 2)) % 99997867;
    }
    cout << (cnt % 99997867) << endl;
}
```
