给定一棵二叉搜索树，请找出其中第k大的节点。

 

```
示例 1:

输入: root = [3,1,4,null,2], k = 1
   3
  / \
 1   4
  \
   2
输出: 4

示例 2:

输入: root = [5,3,6,2,4,null,null,1], k = 3
       5
      / \
     3   6
    / \
   2   4
  /
 1
输出: 4
```
 

#### 限制：

- 1 ≤ k ≤ 二叉搜索树元素个数

#### Solutions

1. ##### inorder traversal with stack

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int kthLargest(TreeNode* root, int k) {
        stack<TreeNode *> s;

        while (root || !s.empty()) {
            while (root) {
                s.push(root);
                root = root->right;
            }
            root = s.top(); s.pop();
            if (--k == 0)
                return root->val;
            else
                root = root->left;
        }

        return 0;
    }
};
```

2. ##### inorder traversal with recursion

```c++
class Solution {
public:
    int curindex = 1;
    bool find = false;
    int kthLargest(TreeNode* root, int k) {
        if (!root)
            return 0;
        int res = kthLargest(root->right, k);
        if (find) return res;
        if (curindex == k) {
            find = true;
            return root->val;
        }
        curindex++;
        return kthLargest(root->left, k);
    }
};
```
